```python
def fetch_index_constituents(force=False):
    """
    Fetch 399975.SZ current constituent stocks via tushare.
    Returns list of dicts with ts_code, name, float_mv etc.
    Cached to data/index_constituents.csv
    """
    path = _cache_path('index_constituents')
    if not force and _is_fresh(path, max_hours=24 * 30):  # monthly freshness
        df = _read_csv(path)
        return df.to_dict('records')

    pro = _get_pro()
    # Try index_weight; fall back to index_member
    try:
        df = pro.index_weight(index_code=INDEX_CODE,
                              trade_date=datetime.now().strftime('%Y%m%d'),
                              fields='index_code,con_code,trade_date,weight')
        if df is not None and not df.empty:
            # Join with stock_basic for name and float_mv
            codes = df['con_code'].unique().tolist()
            basic = pro.stock_basic(ts_code=','.join(codes[:200]),
                                     fields='ts_code,name,list_date,float_mv')
            if basic is not None and not basic.empty:
                df = df.merge(basic, left_on='con_code', right_on='ts_code',
                              how='left', suffixes=('', '_basic'))
                df = df[['con_code', 'name', 'float_mv', 'weight']].rename(
                    columns={'con_code': 'ts_code'})
                df = df.drop_duplicates(subset=['ts_code'])
                df.to_csv(path, index=False)
                return df.to_dict('records')
    except Exception:
        pass
    # Fallback: try index_member
    try:
        df = pro.index_member(index_code=INDEX_CODE,
                              fields='index_code,con_code,con_name,in_date,out_date,is_new')
        if df is not None and not df.empty:
            # Keep current members (is_new == '1')
            active = df[df['is_new'] == '1']
            result = [{'ts_code': r['con_code'], 'name': r.get('con_name', ''),
                       'float_mv': None, 'weight': None}
                      for _, r in active.iterrows()]
            pd.DataFrame(result).to_csv(path, index=False)
            return result
    except Exception:
        pass
    return []
```

```python
"""
Securities sector stock odds screener.
Computes per-cycle Z-scores for index constituent stocks,
aggregates into historical odds rankings.

Usage:
    python screener.py --init           # First run: fetch all + compute
    python screener.py --refresh        # Daily: incremental data + recompute Z
    python screener.py --calc-only      # Recompute Z only (data already fresh)
    python screener.py --refresh-index  # Refetch constituents + recompute thresholds
"""
import os
import sys
import argparse
import pandas as pd
import numpy as np
from datetime import datetime

sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from data_fetcher import (
    fetch_index_constituents, fetch_stock_daily, fetch_index_daily,
    INDEX_CODE, DATA_DIR,
)

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
OUTPUT_DIR = os.path.join(BASE_DIR, 'output')
CYCLES_PATH = os.path.join(OUTPUT_DIR, 'cycles.csv')
ODDS_PATH = os.path.join(OUTPUT_DIR, 'stock_odds.csv')

# Top 3 indicators for signal column
TOP_INDICATORS = [
    ('OBV底背离', {'lookback': 90}),
    ('CCI脱离超卖', {'period': 20}),
    ('MACD柱线缩短', {'fast': 12, 'slow': 26, 'signal': 9, 'consecutive': 3}),
]


def load_cycles():
    """Load bull cycles from output/cycles.csv, 2010+ only."""
    df = pd.read_csv(CYCLES_PATH)
    df['start_date'] = pd.to_datetime(df['start_date'])
    df['end_date'] = pd.to_datetime(df['end_date'])
    df = df[df['start_date'] >= pd.Timestamp('2010-01-01')]
    return df


def ensure_stock_data(ts_code, force=False):
    """Fetch stock daily data, return DataFrame or None."""
    try:
        df = fetch_stock_daily(ts_code, force=force)
        if df is not None and not df.empty:
            df['trade_date'] = pd.to_datetime(df['trade_date'])
            return df.sort_values('trade_date').reset_index(drop=True)
    except Exception:
        pass
    return None


def compute_cycle_return(stock_df, cycle_start, cycle_end):
    """
    Compute max return within a cycle window.
    Returns (max_close / start_close - 1) or None if data insufficient.
    """
    mask = (stock_df['trade_date'] >= cycle_start) & (stock_df['trade_date'] <= cycle_end)
    seg = stock_df[mask]
    if seg.empty or len(seg) < 3:
        return None
    start_price = seg['close'].iloc[0]
    if start_price <= 0:
        return None
    max_price = seg['close'].max()
    return max_price / start_price - 1


def compute_all_z_scores(constituents, cycles, force=False):
    """
    For each constituent and each cycle, compute raw return and Z-score.
    Returns DataFrame with columns: ts_code, cycle_idx, return, z_score
    """
    records = []
    index_df = fetch_index_daily()
    index_df['trade_date'] = pd.to_datetime(index_df['trade_date'])

    for stock in constituents:
        ts_code = stock['ts_code']
        df = ensure_stock_data(ts_code, force=force)
        if df is None:
            continue

        for i, (_, cycle) in enumerate(cycles.iterrows()):
            ret = compute_cycle_return(df, cycle['start_date'], cycle['end_date'])
            if ret is not None:
                # Also compute index return for same window
                idx_ret = compute_cycle_return(index_df, cycle['start_date'], cycle['end_date'])
                records.append({
                    'ts_code': ts_code,
                    'cycle_idx': i,
                    'return': ret,
                    'index_return': idx_ret,
                })

    df = pd.DataFrame(records)
    if df.empty:
        return df

    # Compute Z-scores per cycle (standardize within each cycle's stock pool)
    z_scores = []
    for cycle_idx, group in df.groupby('cycle_idx'):
        returns = group['return'].values
        if len(returns) < 2:
            group['z_score'] = 0.0
        else:
            mu = returns.mean()
            sigma = returns.std(ddof=1)
            if sigma < 1e-8:
                group['z_score'] = 0.0
            else:
                group['z_score'] = (returns - mu) / sigma
        z_scores.append(group)

    return pd.concat(z_scores, ignore_index=True)


def aggregate_odds(detail_df, constituents, existing_meta=None):
    """
    Aggregate per-cycle Z-scores into per-stock odds metrics.
    
    Returns (odds_df, meta) where meta contains M, N, update_time.
    If existing_meta is provided, reuse its M and N thresholds.
    """
    if detail_df.empty:
        return pd.DataFrame(), {}

    agg = detail_df.groupby('ts_code').agg(
        median_z=('z_score', 'median'),
        max_z=('z_score', 'max'),
        positive_z_rate=('z_score', lambda x: (x > 0).sum() / len(x)),
        median_return=('return', 'median'),
        max_return=('return', 'max'),
        cycle_count=('cycle_idx', 'nunique'),
    ).reset_index()

    # Compute beat_index_rate: for each stock, count cycles where stock return > index return
    beat_rates = {}
    for code, group in detail_df.groupby('ts_code'):
        beats = (group['return'] > group['index_return']).sum()
        beat_rates[code] = beats / len(group)
    agg['beat_index_rate'] = agg['ts_code'].map(beat_rates)

    # Add names and float_mv
    name_map = {c['ts_code']: c.get('name', '') for c in constituents}
    mv_map = {c['ts_code']: c.get('float_mv') for c in constituents}
    agg['name'] = agg['ts_code'].map(name_map)
    agg['float_mv'] = agg['ts_code'].map(mv_map)

    # Confidence thresholds
    if existing_meta and 'M' in existing_meta:
        M = existing_meta['M']
    else:
        cycle_counts = agg['cycle_count'].values
        M = int(np.median(cycle_counts)) if len(cycle_counts) > 0 else 3

    def confidence(n):
        if n >= M:
            return '可信'
        elif n >= 2:
            return '参考'
        else:
            return '有限'

    agg['confidence'] = agg['cycle_count'].apply(confidence)

    # Common window N: most recent consecutive cycles with >= 50% participation
    # Count how many stocks participated in each cycle
    total_stocks = len(agg)
    cycle_participation = detail_df.groupby('cycle_idx')['ts_code'].nunique()
    N = 0
    recent_cycles = sorted(cycle_participation.index, reverse=True)
    for ci in recent_cycles:
        if total_stocks > 0 and cycle_participation[ci] / total_stocks >= 0.5:
            N += 1
        else:
            break
    N = max(N, 1)

    meta = {
        'M': M,
        'N': N,
        'update_time': datetime.now().strftime('%Y-%m-%d %H:%M'),
    }

    # Sort by median_z descending
    agg = agg.sort_values('median_z', ascending=False).reset_index(drop=True)
    return agg, meta


def write_odds_csv(agg_df, meta):
    """Write stock_odds.csv with metadata header lines."""
    os.makedirs(OUTPUT_DIR, exist_ok=True)
    with open(ODDS_PATH, 'w') as f:
        f.write(f"# M={meta['M']}, N={meta['N']}, updated={meta['update_time']}\n")
        f.write(f"# M: 置信度阈值（参与轮数中位数）\n")
        f.write(f"# N: 共同窗口轮数（≥50%个股参与的最近连续周期数）\n")
    agg_df.to_csv(ODDS_PATH, mode='a', index=False)
    return ODDS_PATH


def read_odds_csv():
    """Read stock_odds.csv, returning (DataFrame, meta_dict)."""
    meta = {}
    lines = []
    with open(ODDS_PATH, 'r') as f:
        for line in f:
            if line.startswith('#'):
                if 'M=' in line:
                    parts = line.strip('# ').split(',')
                    for p in parts:
                        if '=' in p:
                            k, v = p.split('=')
                            meta[k.strip()] = v.strip()
            else:
                lines.append(line)
    df = pd.read_csv(pd.io.common.StringIO(''.join(lines)))
    return df, meta


def run_init(force=False):
    """Full initialization: fetch constituents + all stock data + compute."""
    print("拉取指数成分股...")
    constituents = fetch_index_constituents(force=True)
    if not constituents:
        print("ERROR: 无法获取成分股列表")
        return
    print(f"  获取到 {len(constituents)} 只成分股")

    cycles = load_cycles()
    print(f"  加载 {len(cycles)} 轮牛市周期（2010年后）")

    print("拉取个股日线数据...")
    detail = compute_all_z_scores(constituents, cycles, force=force)
    print(f"  计算完成: {len(detail)} 条周期记录")

    print("汇总赔率指标...")
    agg, meta = aggregate_odds(detail, constituents)
    print(f"  汇总: {len(agg)} 只个股")

    path = write_odds_csv(agg, meta)
    print(f"输出: {path}")
    print(f"  置信度阈值 M={meta['M']}（参与轮数中位数）")
    print(f"  共同窗口 N={meta['N']}（≥50%参与连续周期数）")
    print(f"  可信: {len(agg[agg['confidence'] == '可信'])} 只")
    print(f"  参考: {len(agg[agg['confidence'] == '参考'])} 只")
    print(f"  有限: {len(agg[agg['confidence'] == '有限'])} 只")


def run_refresh():
    """Incremental data refresh + recompute Z (preserve M, N)."""
    existing_meta = {}
    if os.path.exists(ODDS_PATH):
        _, existing_meta = read_odds_csv()

    constituents = fetch_index_constituents(force=False)
    if not constituents:
        print("ERROR: 无法获取成分股列表")
        return

    cycles = load_cycles()
    detail = compute_all_z_scores(constituents, cycles, force=True)  # force fresh data
    agg, meta = aggregate_odds(detail, constituents, existing_meta=existing_meta)
    path = write_odds_csv(agg, meta)
    print(f"刷新完成: {path}（阈值不变: M={meta['M']}, N={meta['N']}）")


def run_calc_only():

def run_refresh_index():

    print(f"  新阈值: M={meta['M']}, N={meta['N']}")


def get_signal_state(ts_code, stock_df, ref_highs=None):
    """
    Check if a stock currently has buy signals from top 3 indicators.
    Returns dict: {indicator_name: bool}
    """
    from indicators import INDICATOR_REGISTRY, filter_signals

    signals = {}
    for ind_name, params in TOP_INDICATORS:
        cfg = INDICATOR_REGISTRY.get(ind_name)
        if cfg is None:
            signals[ind_name] = False
            continue
        try:
            raw = cfg['func'](stock_df, **params)
            filtered = filter_signals(stock_df, raw, ref_highs=ref_highs)
            # Check last 30 trading days for recent signal
            recent = filtered.tail(30)
            signals[ind_name] = recent.any()
        except Exception:
            signals[ind_name] = False
    return signals
```

```python
def compute_odds_signals():
    """
    Compute buy signals for all stocks in odds_df using top 3 indicators.
    Returns DataFrame with ts_code + signal columns.
    Cached to output/signal_cache.csv until daily data changes.
    """
    cache_path = os.path.join(OUTPUT_DIR, 'signal_cache.csv')
    from indicators import INDICATOR_REGISTRY, filter_signals

    # Check cache freshness: invalid if last_data_date changed
    last_date = st.session_state.get('last_data_date', '')
    cache_valid = False
    if os.path.exists(cache_path) and last_date:
        with open(cache_path, 'r') as f:
            first_line = f.readline()
        if last_date in first_line:
            try:
                cached = pd.read_csv(cache_path)
                if not cached.empty:
                    st.session_state.odds_signal_cache = cached
                    return cached
            except Exception:
                pass

    df = st.session_state.odds_df
    if df is None or df.empty:
        return pd.DataFrame()

    # Use top 3 indicators with their best parameters
    top_inds = [
        ('OBV底背离', {'lookback': 90}),
        ('CCI脱离超卖', {'period': 20}),
        ('MACD柱线缩短', {'fast': 12, 'slow': 26, 'signal': 9, 'consecutive': 3}),
    ]

    records = []
    progress = st.progress(0, '检测各股信号...')
    total = len(df)
    for i, (_, row) in enumerate(df.iterrows()):
        ts_code = row['ts_code']
        code = ts_code.split('.')[0]
        stock_path = os.path.join(DATA_DIR, f'stock_daily_{code}.csv')
        if not os.path.exists(stock_path):
            records.append({
                'ts_code': ts_code,
                'signal': 'no_data',
                'signal_count': 0,
            })
            continue

        sdf = pd.read_csv(stock_path)
        sdf['trade_date'] = pd.to_datetime(sdf['trade_date'])
        sdf = sdf.sort_values('trade_date').reset_index(drop=True)

        sig_count = 0
        for ind_name, params in top_inds:
            cfg = INDICATOR_REGISTRY.get(ind_name)
            if cfg is None:
                continue
            try:
                raw = cfg['func'](sdf, **params)
                filtered = filter_signals(sdf, raw)
                recent = filtered.tail(30)
                if recent.any():
                    sig_count += 1
            except Exception:
                pass

        signal_label = 'buy' if sig_count >= 1 else 'none'
        records.append({
            'ts_code': ts_code,
            'signal': signal_label,
            'signal_count': sig_count,
        })
        progress.progress((i + 1) / total)

    progress.empty()
    result = pd.DataFrame(records)

    # Write cache with date stamp
    with open(cache_path, 'w') as f:
        f.write(f"# cached_date={last_date}\n")
    result.to_csv(cache_path, mode='a', index=False)

    st.session_state.odds_signal_cache = result
    return result
```

```python
def _render_stock_cycle_detail_table(ts_code, odds_df):
    """Show per-cycle detail for a selected stock."""
    cycles = pd.read_csv(os.path.join(OUTPUT_DIR, 'cycles.csv'))
    cycles['start_date'] = pd.to_datetime(cycles['start_date'])
    cycles['end_date'] = pd.to_datetime(cycles['end_date'])
    cycles = cycles[cycles['start_date'] >= pd.Timestamp('2010-01-01')]

    code = ts_code.split('.')[0]
    stock_path = os.path.join(DATA_DIR, f'stock_daily_{code}.csv')
    if not os.path.exists(stock_path):
        st.info("该个股无本地数据")
        return

    sdf = pd.read_csv(stock_path)
    sdf['trade_date'] = pd.to_datetime(sdf['trade_date'])
    sdf = sdf.sort_values('trade_date').reset_index(drop=True)

    idx_path = os.path.join(DATA_DIR, 'index_daily.csv')
    idf = pd.read_csv(idx_path)
    idf['trade_date'] = pd.to_datetime(idf['trade_date'])

    records = []
    for i, (_, cycle) in enumerate(cycles.iterrows()):
        smask = (sdf['trade_date'] >= cycle['start_date']) & (sdf['trade_date'] <= cycle['end_date'])
        imask = (idf['trade_date'] >= cycle['start_date']) & (idf['trade_date'] <= cycle['end_date'])
        sseg = sdf[smask]
        iseg = idf[imask]
        if sseg.empty or iseg.empty or len(sseg) < 3:
            continue
        s_ret = sseg['close'].max() / sseg['close'].iloc[0] - 1
        i_ret = iseg['close'].max() / iseg['close'].iloc[0] - 1
        records.append({
            '周期': f"#{i+1} ({cycle['start_date'].strftime('%Y-%m')})",
            '起始': cycle['start_date'].strftime('%Y-%m-%d'),
            '结束': cycle['end_date'].strftime('%Y-%m-%d'),
            '板块涨幅': f'{i_ret:+.1%}',
            '个股涨幅': f'{s_ret:+.1%}',
            '跑赢': '✓' if s_ret > i_ret else '',
        })

    if records:
        st.dataframe(pd.DataFrame(records), use_container_width=True, hide_index=True)
    else:
        st.info("该个股未参与任何2010年后周期")
```

## 实现修订（实施过程中调整的设计决策）

### 周期定义修订

原本使用 `detect_cycles.py` 自动检测的 9 轮 2010 年后行情。经人工审阅：
- **合并 2014-07~2015-01 与 2015-02~2015-05**：两轮之间仅回撤 4%/30 天，本质为 2014-2015 同一轮大牛市
- **新增 2012-01~2012-06**：从 425 涨至 638（+50%/148 天），被原算法遗漏的独立牛市
- 最终 2010 年后共 9 轮，`cycles.csv` 共 10 行（含 2009 参考）

### filter_signals 状态锁存

原设计 `decline > 20% AND close < MA250` 为即时条件——信号日必须同时满足。实际存在"市场先触底后反弹再启动"的场景，信号日在反弹段，条件已不满足，导致漏判。

修订：
- 保留 `decline > 20%` 和 `close < MA250` 两个条件
- 新增 `state_window=5` 天状态锁存：条件满足后保持 5 个交易日有效
- 实现方式：单遍历时间序列，每个信号日检查之前最近一次条件满足日是否在 5 天内

### 个股 ref_highs 来源

原计划使用指数周期结束日期的个股收盘价作为 ref_highs，后改为**检测个股自己的牛市周期**：
- 使用与指数相同的 `argrelextrema` 算法对每只个股检测周期
- 周期结束日前的时段用 `rolling 250-day max`，周期结束日后用个股自己的周期结束价
- 检测结果缓存至 `data/stock_cycles_{code}.csv`
- 算法参数与指数一致：平滑窗口 20、最低振幅 25%、最短持续 20 天、合并阶数 10

### 数据获取优化

- 成分股列表：tushare `index_weight`/`index_member` 在 2000 积分下不可用，改用用户提供的 49 只硬编码列表（`INDEX_CONSTITUENTS` 字典），同时保留 tushare API 作为优先尝试
- 增量拉取：`fetch_index_daily`/`fetch_stock_daily` 增加增量模式——`force=True` 时读取缓存最新日期，仅从 tushare 拉取之后的新数据并拼接
- API 超时：tushare 客户端设置 10 秒超时，异常时回退缓存数据

### UI 调整

- 行情跟踪"个股详情"下拉列表和"当前信号汇总"表按 Z 评分排序，显示 Z 值
- "各股票最佳指标得分"表按 Z 评分排序
- 指标回测热力图：晚了 ≤5 天（浅黄）与 >5 天（深黄）分色，hover 显示具体天数
- 周期详情 K 线图增加年线（橙色）和跌 20% 线（灰色虚线）
- 个股回测详情 K 线图增加年线；使用个股自己的 ref_highs 计算跌 20% 线

### 重要 Bug 修复

- **个股回测评判公式 Bug**：`weighted_hits` 从未累加，`avg_days`/`weighted_hr` 误缩在 `for` 循环内每轮重写，导致得分仅由 `precision` 决定。修复后与指数回测逻辑一致
- **热力图参数不一致**：`signal_window` 硬编码 45 且未传 `ref_highs` 给 `filter_signals`，与回测引擎不一致。修复为使用 `st.session_state.signal_window` 并构建一致的 ref_highs
- **成分股缓存读取 Bug**：`_read_csv` 使用 `index_col=0` 导致 `ts_code` 列被吞掉。修复为普通 `pd.read_csv`
- **个股详情显示 2009 年行情**：`_render_stock_cycle_detail` 使用了 `cycles_df` 而非 `display_cycles`，修复为 2010 年后行情
