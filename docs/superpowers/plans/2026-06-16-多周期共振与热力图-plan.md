```python
def run_resonance_backtest(cycle_df, standalone_results, data_dfs, signal_window=45):
    """
    Multi-timeframe resonance backtest: weekly indicator fires first,
    then daily indicator confirms within a 30-day observation window.

    Args:
        cycle_df: pd.DataFrame with [start_date, end_date, ...]
        standalone_results: pd.DataFrame from run_backtest (standalone results)
        data_dfs: dict with 'index_daily', 'index_weekly'
        signal_window: int, days to look back before cycle start

    Returns:
        pd.DataFrame with resonance results (same columns as standalone)
    """
    from indicators import INDICATOR_REGISTRY, filter_signals

    cycles = cycle_df.to_dict('records')
    total_cycles = len(cycles)

    # Pick top weekly and daily indicators from standalone results
    weekly_top = standalone_results[standalone_results['周期'] == '周线'].head(3)
    daily_top = standalone_results[standalone_results['周期'] == '日线'].head(3)

    weekly_names = weekly_top['指标名'].unique()
    daily_names = daily_top['指标名'].unique()

    weekly_df = data_dfs['index_weekly'].set_index('trade_date').sort_index() if data_dfs.get('index_weekly') is not None else None
    daily_df = data_dfs['index_daily'].set_index('trade_date').sort_index() if data_dfs.get('index_daily') is not None else None

    if weekly_df is None or daily_df is None:
        return pd.DataFrame()

    results = []

    for w_name in weekly_names:
        w_cfg = INDICATOR_REGISTRY.get(w_name)
        if w_cfg is None:
            continue
        w_params = w_cfg['params'][0]

        # Compute weekly signals (filtered)
        try:
            w_raw = w_cfg['func'](weekly_df.reset_index(), **w_params)
            w_raw.index = weekly_df.index
            w_sig = filter_signals(weekly_df.reset_index(), w_raw)
            w_sig.index = weekly_df.index
        except Exception:
            continue

        w_signal_dates = w_sig[w_sig].index

        for d_name in daily_names:
            d_cfg = INDICATOR_REGISTRY.get(d_name)
            if d_cfg is None:
                continue
            d_params = d_cfg['params'][0]

            try:
                d_raw = d_cfg['func'](daily_df.reset_index(), **d_params)
                d_raw.index = daily_df.index
                d_sig = filter_signals(daily_df.reset_index(), d_raw)
                d_sig.index = daily_df.index
            except Exception:
                continue

            # Build resonance signal series: True on days when both conditions met
            resonance_sig = pd.Series(False, index=daily_df.index)
            total_signals = 0

            for ws_date in w_signal_dates:
                window_end = ws_date + pd.Timedelta(days=30)
                window_mask = (daily_df.index >= ws_date) & (daily_df.index <= window_end)
                window_daily_sigs = d_sig[window_mask]
                if window_daily_sigs.any():
                    # Mark the first daily signal in the window
                    first_sig_date = window_daily_sigs[window_daily_sigs].index[0]
                    resonance_sig[first_sig_date] = True
                    total_signals += 1

            # Judge each cycle
            hits = 0
            lates = 0
            days_list = []
            hit_signal_count = 0

            for cycle in cycles:
                start = pd.Timestamp(cycle['start_date'])
                end = pd.Timestamp(cycle['end_date'])
                window_start = start - pd.Timedelta(days=signal_window)
                window_end = start + pd.Timedelta(days=10)

                window_mask = (resonance_sig.index >= window_start) & (resonance_sig.index <= window_end)
                if resonance_sig[window_mask].any():
                    hit_signal_count += 1

                judgment = judge_signal(resonance_sig, start, end, signal_window)
                if judgment['hit']:
                    hits += 1
                    days_list.append(judgment['days_before'])
                elif judgment['late']:
                    lates += 1

            hit_rate = hits / total_cycles if total_cycles > 0 else 0
            avg_days = np.mean(days_list) if days_list else None

            precision = hit_signal_count / total_signals if total_signals > 0 else 0
            days_score = 0
            if avg_days is not None:
                days_score = abs(avg_days - 5) / 30 * 0.3
            score = hit_rate * 0.4 + precision * 0.3 - days_score

            rule_name = f'周线:{w_name} + 日线:{d_name}'
            results.append({
                '指标名': rule_name,
                '类别': '共振',
                '周期': '周+日',
                '参数': f'w={w_params}, d={d_params}',
                '命中率': round(hit_rate, 3),
                '命中轮数': f'{hits}/{total_cycles}',
                '平均提前天': round(avg_days, 1) if avg_days is not None else None,
                '信号有效率': round(precision, 3),
                '总信号数': total_signals,
                '综合得分': round(score, 3),
            })

    return pd.DataFrame(results)


def run_and_save_all(data_dfs, signal_window=45):
    """Run standalone + resonance backtest, save merged results."""
    standalone = run_and_save(data_dfs, signal_window)
    resonance = run_resonance_backtest(
        pd.read_csv(os.path.join(OUTPUT_DIR, 'cycles.csv')),
        standalone, data_dfs, signal_window,
    )
    if not resonance.empty:
        combined = pd.concat([standalone, resonance], ignore_index=True)
        combined.to_csv(os.path.join(OUTPUT_DIR, 'backtest_results.csv'), index=False, encoding='utf-8-sig')
        print(f"Total results: {len(standalone)} standalone + {len(resonance)} resonance = {len(combined)}")
        return combined
    return standalone
```

```python
def _render_heatmap(results_df, cycles_df):
    """Render indicator × cycle heatmap using Plotly."""
    cycles = cycles_df.to_dict('records')
    # Filter: exclude 资金 category, exclude resonance rows for standalone heatmap
    heat_df = results_df[results_df['类别'] != '共振'].copy()
    heat_df = heat_df.sort_values('综合得分', ascending=False)

    # Build z-matrix: rows=indicators, cols=cycles
    z_data = []     # 0=miss(red), 1=late(yellow), 2=hit-far(light green), 3=hit-near(dark green)
    z_text = []     # hover text
    y_labels = []

    for _, row in heat_df.iterrows():
        y_labels.append(f"{row['指标名']}({row['周期']})")
        z_row = []
        t_row = []

        # Recompute signals for this indicator across all cycles
        freq = row['周期']
        if '周' in freq and st.session_state.index_weekly is not None:
            df = st.session_state.index_weekly.copy()
        else:
            df = st.session_state.index_daily.copy()
        df['trade_date'] = pd.to_datetime(df['trade_date'])
        df = df.set_index('trade_date').sort_index()

        from indicators import INDICATOR_REGISTRY, filter_signals
        import ast
        cfg = INDICATOR_REGISTRY.get(row['指标名'])
        if cfg is None:
            continue
        try:
            params_str = row['参数']
            if isinstance(params_str, str):
                params = ast.literal_eval(params_str)
            else:
                params = cfg['params'][0]
        except Exception:
            params = cfg['params'][0]

        try:
            sig_raw = cfg['func'](df.reset_index(), **params)
            sig_raw.index = df.index
            sig = filter_signals(df.reset_index(), sig_raw)
            sig.index = df.index
        except Exception:
            for _ in cycles:
                z_row.append(0)
                t_row.append('error')
            z_data.append(z_row)
            z_text.append(t_row)
            continue

        from backtest import judge_signal
        for cycle in cycles:
            start = pd.Timestamp(cycle['start_date'])
            end = pd.Timestamp(cycle['end_date'])
            j = judge_signal(sig, start, end, signal_window=45)
            if j['hit']:
                days = j['days_before']
                if days <= 15:
                    z_row.append(3)  # dark green
                else:
                    z_row.append(2)  # light green
                t_row.append(f"命中 提前{days}天")
            elif j['late']:
                z_row.append(1)  # yellow
                t_row.append('太晚')
            else:
                z_row.append(0)  # red
                t_row.append('错过')

        z_data.append(z_row)
        z_text.append(t_row)

    if not z_data:
        st.info("无数据可显示")
        return

    x_labels = [f"{c['start_date'][:10]}\n+{c['change_pct']}%" for c in cycles]

    colorscale = [
        [0.0, '#ef5350'],   # red: missed
        [0.33, '#ffb74d'],  # yellow/orange: late
        [0.66, '#a5d6a7'],  # light green: hit (far)
        [1.0, '#2e7d32'],   # dark green: hit (near)
    ]

    fig = go.Figure(data=go.Heatmap(
        z=z_data,
        text=z_text,
        x=x_labels,
        y=y_labels,
        colorscale=colorscale,
        zmin=0, zmax=3,
        showscale=False,
        hovertemplate='%{y}<br>%{x}<br>%{text}<extra></extra>',
    ))

    fig.update_layout(
        height=max(300, len(y_labels) * 25 + 100),
        margin=dict(l=10, r=10, t=20, b=60),
        xaxis=dict(side='bottom', tickangle=0),
        yaxis=dict(autorange='reversed'),
    )

    st.plotly_chart(fig, use_container_width=True)
```
