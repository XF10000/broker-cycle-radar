```python
"""Shared tushare data fetching with local CSV caching."""
import os
import pandas as pd
import tushare as ts
from datetime import datetime, timedelta

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
DATA_DIR = os.path.join(BASE_DIR, 'data')
TOKEN_FILE = os.path.join(BASE_DIR, '.tushare_token')

STOCKS = {
    '601066': '中信建投',
    '600030': '中信证券',
    '600918': '中泰证券',
    '601696': '中银证券',
    '600906': '财达证券',
}

INDEX_CODE = '399975.SZ'


def _get_pro():
    with open(TOKEN_FILE) as f:
        token = f.read().strip()
    ts.set_token(token)
    return ts.pro_api()


def _cache_path(name):
    os.makedirs(DATA_DIR, exist_ok=True)
    return os.path.join(DATA_DIR, f'{name}.csv')


def _is_fresh(path, max_hours=12):
    if not os.path.exists(path):
        return False
    mtime = datetime.fromtimestamp(os.path.getmtime(path))
    return (datetime.now() - mtime).total_seconds() < max_hours * 3600


def _read_csv(path):
    df = pd.read_csv(path, index_col=0)
    if 'trade_date' in df.columns:
        df['trade_date'] = pd.to_datetime(df['trade_date'])
    return df


def fetch_index_daily(force=False):
    """Fetch 399975.SZ daily K-line. Cached to data/index_daily.csv"""
    path = _cache_path('index_daily')
    if not force and _is_fresh(path):
        return _read_csv(path)

    pro = _get_pro()
    df = pro.index_daily(ts_code=INDEX_CODE, start_date='20100101',
                         fields='ts_code,trade_date,open,high,low,close,vol,amount')
    df['trade_date'] = pd.to_datetime(df['trade_date'])
    df = df.sort_values('trade_date').reset_index(drop=True)
    df.to_csv(path)
    return df


def fetch_stock_daily(ts_code, force=False):
    """Fetch individual stock daily K-line."""
    code = ts_code.split('.')[0]
    path = _cache_path(f'stock_daily_{code}')
    if not force and _is_fresh(path):
        return _read_csv(path)

    pro = _get_pro()
    df = pro.daily(ts_code=ts_code, start_date='20100101',
                   fields='ts_code,trade_date,open,high,low,close,vol,amount')
    if df is not None and not df.empty:
        df['trade_date'] = pd.to_datetime(df['trade_date'])
        df = df.sort_values('trade_date').reset_index(drop=True)
        df.to_csv(path)
        return df
    return pd.DataFrame()


def fetch_all_stocks_daily(force=False):
    """Fetch all 5 candidate stocks daily data."""
    result = {}
    for code, name in STOCKS.items():
        ts_code = f'{code}.SH' if code.startswith('6') else f'{code}.SZ'
        df = fetch_stock_daily(ts_code, force)
        if not df.empty:
            result[code] = df
    return result


def daily_to_weekly(df):
    """Convert daily DataFrame to weekly by resampling."""
    df = df.copy()
    df['trade_date'] = pd.to_datetime(df['trade_date'])
    df = df.set_index('trade_date')
    weekly = df.resample('W').agg({
        'open': 'first',
        'high': 'max',
        'low': 'min',
        'close': 'last',
        'vol': 'sum',
        'amount': 'sum',
    }).dropna().reset_index()
    return weekly


def fetch_moneyflow(ts_code, force=False):
    """
    Fetch moneyflow data for a stock.
    Returns empty DataFrame if permission denied or API unavailable.
    """
    code = ts_code.split('.')[0]
    path = _cache_path(f'moneyflow_{code}')
    if not force and _is_fresh(path):
        df = _read_csv(path)
        if 'trade_date' in df.columns:
            return df
    try:
        pro = _get_pro()
        df = pro.moneyflow_dc(ts_code=ts_code, start_date='20100101')
        if df is not None and not df.empty:
            df['trade_date'] = pd.to_datetime(df['trade_date'])
            df = df.sort_values('trade_date').reset_index(drop=True)
            df.to_csv(path)
            return df
    except Exception:
        pass
    return pd.DataFrame()
```

```python
"""
One-time cycle detection for securities sector index (399975.SZ).
Detects bull market cycles using trough-to-peak analysis on smoothed price.

Usage: python detect_cycles.py
Output: output/cycles.csv
"""
import os
import numpy as np
import pandas as pd
from scipy.signal import argrelextrema

from data_fetcher import fetch_index_daily

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
OUTPUT_DIR = os.path.join(BASE_DIR, 'output')

# ---- Tunable constants ----
SMOOTH_WINDOW = 20       # MA period for smoothing
MIN_AMPLITUDE_PCT = 15   # Minimum rally amplitude (%)
MIN_DURATION_DAYS = 20   # Minimum rally duration (trading days)
MERGE_ORDER = 10          # Order for argrelextrema (half-window size)
# ---------------------------


def detect_cycles(df, smooth_window=SMOOTH_WINDOW,
                  min_amp_pct=MIN_AMPLITUDE_PCT,
                  min_days=MIN_DURATION_DAYS):
    """
    Detect bull cycles from price data.

    Returns list of dicts: [{start_date, end_date, start_price, end_price, change_pct, duration_days}]
    """
    close = df['close'].values
    dates = df['trade_date'].values

    # Smooth the close price
    smoothed = pd.Series(close).rolling(smooth_window, min_periods=1).mean().values

    # Find local minima (troughs) and maxima (peaks)
    troughs = argrelextrema(smoothed, np.less, order=MERGE_ORDER)[0]
    peaks = argrelextrema(smoothed, np.greater, order=MERGE_ORDER)[0]

    # Build cycles: each trough followed by next peak = one bull cycle
    cycles = []
    for t_idx in troughs:
        # Find the next peak after this trough
        later_peaks = peaks[peaks > t_idx]
        if len(later_peaks) == 0:
            continue
        p_idx = later_peaks[0]

        duration = int(p_idx - t_idx)
        if duration < min_days:
            continue

        start_price = close[t_idx]
        end_price = close[p_idx]
        change_pct = (end_price - start_price) / start_price * 100

        if change_pct < min_amp_pct:
            continue

        cycles.append({
            'start_date': pd.Timestamp(dates[t_idx]),
            'end_date': pd.Timestamp(dates[p_idx]),
            'start_price': round(start_price, 2),
            'end_price': round(end_price, 2),
            'change_pct': round(change_pct, 2),
            'duration_days': duration,
        })

    # Remove overlapping cycles (keep the one with larger amplitude when overlap)
    filtered = []
    for c in cycles:
        if not filtered:
            filtered.append(c)
            continue
        last = filtered[-1]
        if c['start_date'] <= last['end_date']:
            # Overlap: keep the one with larger amplitude
            if c['change_pct'] > last['change_pct']:
                filtered[-1] = c
        else:
            filtered.append(c)

    return filtered


def main():
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    print("Fetching index data...")
    df = fetch_index_daily()
    print(f"Loaded {len(df)} daily bars from {df['trade_date'].iloc[0].date()} to {df['trade_date'].iloc[-1].date()}")

    print(f"\nDetecting cycles (smooth={SMOOTH_WINDOW}, min_amp={MIN_AMPLITUDE_PCT}%, min_days={MIN_DURATION_DAYS})...")
    cycles = detect_cycles(df)

    print(f"\nFound {len(cycles)} bull cycles:\n")
    print(f"{'#':<4} {'Start':<12} {'End':<12} {'Days':<6} {'Start$':<8} {'End$':<8} {'Change%':<8}")
    print("-" * 68)
    for i, c in enumerate(cycles, 1):
        print(f"{i:<4} {str(c['start_date'].date()):<12} {str(c['end_date'].date()):<12} "
              f"{c['duration_days']:<6} {c['start_price']:<8} {c['end_price']:<8} {c['change_pct']:<8}")

    # Save to CSV
    cycles_df = pd.DataFrame(cycles)
    path = os.path.join(OUTPUT_DIR, 'cycles.csv')
    cycles_df.to_csv(path, index=False)
    print(f"\nSaved to {path}")


if __name__ == '__main__':
    main()
```

```python
"""
Technical indicator library for securities sector bull-run signal detection.
All signals are early-warning type: divergence > crossover, turning > breakout.

Each indicator function:
    Input:  df (pd.DataFrame with columns: trade_date, open, high, low, close, vol, amount)
    Output: pd.Series of bool (True = signal triggered on that day)
"""
import numpy as np
import pandas as pd
from itertools import product


# ============================================================
# Helpers
# ============================================================

def _ema(s, span):
    return s.ewm(span=span, adjust=False).mean()

def _sma(s, window):
    return s.rolling(window=window, min_periods=1).mean()

def _rolling_min_idx(arr, window):
    """Return index (within window) of minimum for each position. Uses expanding for first window-1 positions."""
    result = np.zeros(len(arr), dtype=int)
    for i in range(len(arr)):
        start = max(0, i - window + 1)
        result[i] = i - start - np.argmin(arr[start:i+1][::-1])
    return result


# ============================================================
# 5.1 Trend indicators
# ============================================================

def macd_divergence(df, fast=12, slow=26, signal=9, lookback=60):
    """
    MACD bottom divergence: price makes new low in lookback window
    but DIF is higher than its prior low. Also: histogram bars shortening 3+ days.
    """
    close = df['close']
    dif = _ema(close, fast) - _ema(close, slow)
    dea = _ema(dif, signal)
    hist = 2 * (dif - dea)

    sig = pd.Series(False, index=df.index)
    c = close.values
    d = dif.values
    h = hist.values

    # Divergence
    for i in range(lookback, len(df)):
        wc = c[i-lookback:i+1]
        wd = d[i-lookback:i+1]
        if wc[-1] <= wc.min() and wd[-1] > wd.min():
            sig.iloc[i] = True

    # Histogram shortening 3+ consecutive days while still negative
    for i in range(3, len(df)):
        if (h[i-2] < h[i-1] < h[i]) and h[i-2] < 0 and h[i-1] < 0:
            sig.iloc[i] = True

    return sig


def ma_return(df, ma_period=60, below_days=20):
    """
    Price crosses back above MA after running below it for at least below_days.
    """
    ma = _sma(df['close'], ma_period)
    below = df['close'] < ma

    sig = pd.Series(False, index=df.index)
    consec = below.astype(int).groupby((below != below.shift()).cumsum()).cumsum()

    for i in range(1, len(df)):
        if consec.iloc[i-1] >= below_days and below.iloc[i-1] and not below.iloc[i]:
            sig.iloc[i] = True
    return sig


def adx_turn(df, period=14, adx_low=20):
    """
    ADX turns up from below adx_low, with +DI crossing above -DI simultaneously.
    """
    high, low, close = df['high'], df['low'], df['close']

    tr = pd.concat([
        high - low,
        (high - close.shift()).abs(),
        (low - close.shift()).abs()
    ], axis=1).max(axis=1)
    atr = _sma(tr, period)

    up = high.diff()
    dn = -low.diff()
    p_dm = pd.Series(np.where((up > dn) & (up > 0), up, 0), index=df.index)
    m_dm = pd.Series(np.where((dn > up) & (dn > 0), dn, 0), index=df.index)

    p_di = 100 * _sma(p_dm, period) / atr
    m_di = 100 * _sma(m_dm, period) / atr
    dx = 100 * (p_di - m_di).abs() / (p_di + m_di + 1e-10)
    adx = _sma(dx, period)

    sig = pd.Series(False, index=df.index)
    for i in range(period*2, len(df)):
        if (adx.iloc[i] > adx.iloc[i-1] and
            adx.iloc[i-1] <= adx_low and
            p_di.iloc[i] > m_di.iloc[i] and
            p_di.iloc[i-1] <= m_di.iloc[i-1]):
            sig.iloc[i] = True
    return sig


# ============================================================
# 5.2 Momentum indicators
# ============================================================

def rsi_divergence(df, period=14, lookback=60):
    """
    RSI bottom divergence: price at new low in lookback but RSI is not.
    Also: RSI forms W-bottom in oversold zone (< 30).
    """
    close = df['close']
    delta = close.diff()
    gain = delta.clip(lower=0)
    loss = (-delta).clip(lower=0)
    avg_gain = _sma(gain, period)
    avg_loss = _sma(loss, period)
    rs = avg_gain / (avg_loss + 1e-10)
    rsi = 100 - (100 / (1 + rs))

    sig = pd.Series(False, index=df.index)
    c = close.values
    r = rsi.values

    for i in range(lookback, len(df)):
        wc = c[i-lookback:i+1]
        wr = r[i-lookback:i+1]
        if (wc[-1] <= wc.min() and          # price at new low in window
            wr[-1] > wr.min() * 0.95 and     # RSI NOT at new low
            wr[-1] < 50):                    # RSI not already overbought
            sig.iloc[i] = True

    return sig


def kdj_j_reversal(df, n=9, m1=3, m2=3):
    """
    KDJ: J value crosses above +20 after being below 0.
    This is earlier than K/D golden cross.
    """
    low_n = df['low'].rolling(n).min()
    high_n = df['high'].rolling(n).max()
    rsv = (df['close'] - low_n) / (high_n - low_n + 1e-10) * 100

    k = rsv.ewm(alpha=1/m1, adjust=False).mean()
    d = k.ewm(alpha=1/m2, adjust=False).mean()
    j = 3 * k - 2 * d

    sig = pd.Series(False, index=df.index)
    for i in range(2, len(df)):
        if j.iloc[i-1] < 0 and j.iloc[i] > 20:
            sig.iloc[i] = True
    return sig


def cci_reversal(df, period=20):
    """
    CCI crosses above -100 from below.
    """
    tp = (df['high'] + df['low'] + df['close']) / 3
    ma = _sma(tp, period)
    md = tp.rolling(period).apply(lambda x: np.abs(x - x.mean()).mean(), raw=True)
    cci = (tp - ma) / (0.015 * md + 1e-10)

    sig = pd.Series(False, index=df.index)
    for i in range(1, len(df)):
        if cci.iloc[i-1] < -100 and cci.iloc[i] > -100:
            sig.iloc[i] = True
    return sig


# ============================================================
# 5.3 Volume indicators
# ============================================================

def volume_contraction(df, ma_period=20, vol_ratio=0.5, flat_days=5, flat_pct=2.0):
    """
    Volume shrinks below ratio * MA, and price stays flat (range < flat_pct%).
    Signal fires on the last day of the flat period.
    """
    vol_ma = _sma(df['vol'], ma_period)
    low_vol = df['vol'] < vol_ma * vol_ratio

    close = df['close'].values
    sig = pd.Series(False, index=df.index)

    for i in range(flat_days, len(df)):
        if not low_vol.iloc[i]:
            continue
        window = close[i-flat_days:i+1]
        price_range = (window.max() - window.min()) / window.mean() * 100
        if price_range < flat_pct:
            sig.iloc[i] = True
    return sig


def moderate_expansion(df, contract_window=10, expand_ratio=1.3, max_ratio=1.5):
    """
    After a period of low volume, first day with moderately expanded volume + small bullish candle.
    """
    vol_ma = _sma(df['vol'], 20)
    close = df['close']
    open_ = df['open']

    sig = pd.Series(False, index=df.index)

    for i in range(contract_window + 1, len(df)):
        # Check prior window had below-average volume
        prior_vol = df['vol'].iloc[i-contract_window:i]
        if prior_vol.mean() > vol_ma.iloc[i] * 0.8:
            continue
        # Today's volume moderately above MA
        ratio = df['vol'].iloc[i] / vol_ma.iloc[i]
        if ratio < expand_ratio or ratio > max_ratio:
            continue
        # Small bullish candle (1% to 3% gain)
        chg = (close.iloc[i] - close.iloc[i-1]) / close.iloc[i-1] * 100
        if 1.0 <= chg <= 3.0 and close.iloc[i] > open_.iloc[i]:
            sig.iloc[i] = True
    return sig


def obv_divergence(df, lookback=60):
    """
    OBV (On-Balance Volume) divergence: price sideways or declining
    but OBV trending upward.
    """
    close = df['close']
    vol = df['vol']

    obv = pd.Series(0.0, index=df.index)
    for i in range(1, len(df)):
        if close.iloc[i] > close.iloc[i-1]:
            obv.iloc[i] = obv.iloc[i-1] + vol.iloc[i]
        elif close.iloc[i] < close.iloc[i-1]:
            obv.iloc[i] = obv.iloc[i-1] - vol.iloc[i]
        else:
            obv.iloc[i] = obv.iloc[i-1]

    sig = pd.Series(False, index=df.index)
    c = close.values
    o = obv.values

    for i in range(lookback, len(df)):
        wc = c[i-lookback:i+1]
        wo = o[i-lookback:i+1]
        if wc[-1] <= wc.mean() and wo[-1] > wo.mean():
            sig.iloc[i] = True
    return sig


# ============================================================
# 5.4 Volatility indicators
# ============================================================

def bollinger_squeeze(df, period=20, std=2, ndays_newlow=60):
    """
    Bollinger Band width at N-day low — signals impending breakout.
    """
    ma = _sma(df['close'], period)
    stddev = df['close'].rolling(period).std()
    upper = ma + std * stddev
    lower = ma - std * stddev
    bandwidth = (upper - lower) / (ma + 1e-10)

    sig = pd.Series(False, index=df.index)
    for i in range(ndays_newlow, len(df)):
        if bandwidth.iloc[i] <= bandwidth.iloc[i-ndays_newlow:i].min():
            sig.iloc[i] = True
    return sig


def bollinger_lower_rebound(df, period=20, std=2):
    """
    Price crosses above middle band after running along or below lower band.
    """
    ma = _sma(df['close'], period)
    stddev = df['close'].rolling(period).std()
    lower = ma - std * stddev

    sig = pd.Series(False, index=df.index)
    for i in range(period, len(df)):
        # Price was at or below lower band recently, now crosses above middle
        if (df['close'].iloc[i-1] < ma.iloc[i-1] and
            df['close'].iloc[i] > ma.iloc[i] and
            (df['low'].iloc[i-3:i].min() <= lower.iloc[i-3:i].max())):
            sig.iloc[i] = True
    return sig


# ============================================================
# 5.5 Money flow indicators
# ============================================================

def main_force_inflow(df, consecutive_days=3):
    """
    Consecutive days of positive net main force inflow.
    Requires 'buy_elg_vol' and 'sell_elg_vol' columns in df (from moneyflow data).
    Returns empty signal series if columns missing.
    """
    sig = pd.Series(False, index=df.index)
    if 'buy_elg_vol' not in df.columns or 'sell_elg_vol' not in df.columns:
        return sig

    net_inflow = df['buy_elg_vol'] - df['sell_elg_vol']
    positive = net_inflow > 0

    for i in range(consecutive_days - 1, len(df)):
        if positive.iloc[i-consecutive_days+1:i+1].all():
            sig.iloc[i] = True
    return sig


def margin_stabilize(df, window=5, threshold=-1.0):
    """
    Margin balance change rate > threshold% (stopped declining).
    Requires 'rzye' column (margin balance). Returns empty if missing.
    """
    sig = pd.Series(False, index=df.index)
    if 'rzye' not in df.columns:
        return sig

    change_pct = df['rzye'].pct_change(periods=window) * 100

    for i in range(window + 1, len(df)):
        if change_pct.iloc[i] > threshold:
            sig.iloc[i] = True
    return sig
```

```python
"""
Backtest engine: evaluates technical indicator signals against known market cycles.
"""
import os
import pandas as pd
import numpy as np
from itertools import product

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
OUTPUT_DIR = os.path.join(BASE_DIR, 'output')


def judge_signal(signal_series, cycle_start, cycle_end, signal_window=30):
    """
    Judge whether a signal successfully predicted a market cycle.

    Args:
        signal_series: pd.Series of bool, indexed by date
        cycle_start: pd.Timestamp, the start date of the bull cycle
        cycle_end: pd.Timestamp, the end date of the bull cycle
        signal_window: int, days before cycle_start to look for signals

    Returns:
        dict with keys: hit (bool), signal_date (pd.Timestamp or None),
        days_before (int or None), late (bool)
    """
    # Valid signal window: [cycle_start - signal_window, cycle_start]
    window_start = cycle_start - pd.Timedelta(days=signal_window)
    window_mask = (signal_series.index >= window_start) & (signal_series.index <= cycle_start)
    window_signals = signal_series[window_mask]

    if window_signals.any():
        # Take the latest signal in the window (closest to cycle_start)
        signal_dates = window_signals[window_signals].index
        signal_date = signal_dates[-1]
        days_before = (cycle_start - signal_date).days
        return {
            'hit': True,
            'signal_date': signal_date,
            'days_before': days_before,
            'late': False,
        }

    # Check if signal fires too late (after cycle_start + 10 days)
    late_cutoff = cycle_start + pd.Timedelta(days=10)
    late_mask = (signal_series.index > cycle_start) & (signal_series.index <= late_cutoff)
    if signal_series[late_mask].any():
        return {
            'hit': False,
            'signal_date': signal_series[late_mask].index[0],
            'days_before': None,
            'late': True,
        }

    return {
        'hit': False,
        'signal_date': None,
        'days_before': None,
        'late': False,
    }


def count_false_signals(signal_series, cycles, signal_window=30):
    """
    Count signals that fall outside any cycle's valid window.
    A signal is 'false' if it's not in any [start-signal_window, start+10] window.
    """
    # Build union of all valid+late windows
    valid_mask = pd.Series(False, index=signal_series.index)
    for cycle in cycles:
        start = pd.Timestamp(cycle['start_date'])
        end = pd.Timestamp(cycle['end_date'])
        window_start = start - pd.Timedelta(days=signal_window)
        window_end = start + pd.Timedelta(days=10)
        valid_mask[(signal_series.index >= window_start) & (signal_series.index <= window_end)] = True

    false_signals = signal_series & ~valid_mask
    return false_signals.sum()


def run_backtest(cycle_df, signals_list, data_dfs, signal_window=30):
    """
    Run backtest for all signal rules against all cycles.

    Args:
        cycle_df: pd.DataFrame with columns [start_date, end_date, ...]
        signals_list: list of dicts from indicators.get_all_signal_rules()
        data_dfs: dict with keys 'index_daily', 'index_weekly', and moneyflow data
        signal_window: int, days to look back before cycle start

    Returns:
        pd.DataFrame with backtest results
    """
    cycles = cycle_df.to_dict('records')
    total_cycles = len(cycles)

    results = []

    for freq_key, df in [('日线', data_dfs.get('index_daily')),
                          ('周线', data_dfs.get('index_weekly'))]:
        if df is None or df.empty:
            continue

        df = df.set_index('trade_date').sort_index()
        for freq_key_mf, mf_df in [('日线', data_dfs.get('index_moneyflow')),
                                    (None, None)]:
            pass  # Moneyflow is per-stock, handled per-rule below

        for rule in signals_list:
            # Skip weekly for some indicators that work better on daily
            # Moneyflow indicators: use merged df if available
            work_df = df.copy()

            needs_mf = rule.get('requires_moneyflow', False)
            if needs_mf:
                mf_df = data_dfs.get('index_moneyflow')
                if mf_df is None or mf_df.empty:
                    continue  # Skip moneyflow rules if no data
                mf_df = mf_df.set_index('trade_date').sort_index()
                # Merge moneyflow columns into work_df
                for col in mf_df.columns:
                    if col not in work_df.columns:
                        work_df[col] = mf_df[col]

            try:
                signal_series = rule['func'](work_df.reset_index(), **rule['params'])
                signal_series.index = work_df.index
            except Exception as e:
                print(f"  [SKIP] {rule['name']} ({freq_key}): {e}")
                continue

            # Judge each cycle
            hits = 0
            lates = 0
            days_list = []
            for cycle in cycles:
                judgment = judge_signal(
                    signal_series,
                    pd.Timestamp(cycle['start_date']),
                    pd.Timestamp(cycle['end_date']),
                    signal_window,
                )
                if judgment['hit']:
                    hits += 1
                    days_list.append(judgment['days_before'])
                elif judgment['late']:
                    lates += 1

            hit_rate = hits / total_cycles if total_cycles > 0 else 0
            avg_days = np.mean(days_list) if days_list else None
            false_count = count_false_signals(signal_series, cycles, signal_window)
            false_rate = false_count / total_cycles if total_cycles > 0 else 0

            # Composite score
            days_score = 0
            if avg_days is not None:
                days_score = abs(avg_days - 5) / 30 * 0.2
            score = hit_rate * 0.5 - false_rate * 0.3 - days_score

            results.append({
                '指标名': rule['name'],
                '类别': rule['category'],
                '周期': freq_key,
                '参数': str(rule['params']),
                '命中率': round(hit_rate, 3),
                '命中轮数': f'{hits}/{total_cycles}',
                '平均提前天': round(avg_days, 1) if avg_days is not None else None,
                '假信号率': round(false_rate, 2),
                '假信号数': int(false_count),
                '综合得分': round(score, 3),
            })

    results_df = pd.DataFrame(results)
    results_df = results_df.sort_values('综合得分', ascending=False).reset_index(drop=True)
    return results_df


def run_and_save(data_dfs, signal_window=30):
    """Convenience: run backtest and save to CSV."""
    from indicators import get_all_signal_rules

    cycle_path = os.path.join(OUTPUT_DIR, 'cycles.csv')
    if not os.path.exists(cycle_path):
        raise FileNotFoundError(f"{cycle_path} not found. Run detect_cycles.py first.")

    cycle_df = pd.read_csv(cycle_path)
    rules = get_all_signal_rules()
    results = run_backtest(cycle_df, rules, data_dfs, signal_window)

    os.makedirs(OUTPUT_DIR, exist_ok=True)
    path = os.path.join(OUTPUT_DIR, 'backtest_results.csv')
    results.to_csv(path, index=False, encoding='utf-8-sig')
    print(f"Saved {len(results)} results to {path}")
    return results
```

```python
    indicator_names = filtered['指标名'].unique()
    selected_indicator = st.selectbox('选择指标', indicator_names)

    if selected_indicator:
        subset = filtered[filtered['指标名'] == selected_indicator]
        if not subset.empty and st.session_state.index_daily is not None and st.session_state.cycles_df is not None:
            # Show the indicator on all cycles
            cycles = st.session_state.cycles_df.to_dict('records')
            n_cycles = len(cycles)
            n_cols = min(3, n_cycles)
            n_rows = (n_cycles + n_cols - 1) // n_cols

            for row_idx in range(n_rows):
                cols = st.columns(n_cols)
                for col_idx in range(n_cols):
                    cycle_idx = row_idx * n_cols + col_idx
                    if cycle_idx >= n_cycles:
                        break
                    cycle = cycles[cycle_idx]
                    with cols[col_idx]:
                        _render_cycle_detail(cycle, cycle_idx, selected_indicator)


def _render_cycle_detail(cycle, cycle_idx, indicator_name):
    """Render a small chart for one cycle showing indicator signals."""
    start = pd.Timestamp(cycle['start_date'])
    end = pd.Timestamp(cycle['end_date'])

    # Expand view window: 60 days before start to end of cycle
    view_start = start - pd.Timedelta(days=90)
    view_end = end + pd.Timedelta(days=20)

    df = st.session_state.index_daily.copy()
    df['trade_date'] = pd.to_datetime(df['trade_date'])
    mask = (df['trade_date'] >= view_start) & (df['trade_date'] <= view_end)
    segment = df[mask].reset_index(drop=True)

    if segment.empty:
        st.caption(f"行情{cycle_idx+1}: 无数据")
        return

    # Get indicator signal
    cfg = INDICATOR_REGISTRY.get(indicator_name)
    if cfg is None:
        return

    params = cfg['params'][0]  # Use first param set for display
    try:
        signals = cfg['func'](segment, **params)
    except Exception:
        return

    # Build chart
    fig = make_subplots(
        rows=2, cols=1,
        shared_xaxes=True,
        vertical_spacing=0.03,
        row_heights=[0.7, 0.3],
    )

    fig.add_trace(go.Candlestick(
        x=segment['trade_date'],
        open=segment['open'], high=segment['high'],
        low=segment['low'], close=segment['close'],
        name='价格',
        increasing_line_color='red',
        decreasing_line_color='green',
    ), row=1, col=1)

    # Mark signal dates
    if signals is not None:
        sig_dates = segment['trade_date'][signals.values]
        sig_closes = segment['close'][signals.values]
        if len(sig_dates) > 0:
            fig.add_trace(go.Scatter(
                x=sig_dates, y=sig_closes,
                mode='markers',
                marker=dict(symbol='triangle-up', size=12, color='blue'),
                name='信号',
            ), row=1, col=1)

    # Add volume bar
    colors = ['red' if segment['close'].iloc[i] >= segment['open'].iloc[i] else 'green'
              for i in range(len(segment))]
    fig.add_trace(go.Bar(
        x=segment['trade_date'], y=segment['vol'],
        name='成交量', marker_color=colors,
        opacity=0.5,
    ), row=2, col=1)

    # Mark cycle start
    fig.add_vline(x=start, line_dash='dash', line_color='purple',
                  annotation_text='启动', row=1, col=1)

    fig.update_layout(
        title=f"行情{cycle_idx+1}: {start.date()} → {end.date()} (+{cycle['change_pct']}%)",
        height=350,
        showlegend=False,
        margin=dict(l=10, r=10, t=40, b=10),
    )
    fig.update_xaxes(type='date', rangeslider_visible=False)

    st.plotly_chart(fig, use_container_width=True)

def render_stock_verification():
    st.header("个股验证")

    if not st.session_state.stocks_daily:
        st.warning("请先在侧边栏加载数据")
        return

    # Stock selector
    stock_codes = list(st.session_state.stocks_daily.keys())
    stock_names = {code: STOCKS.get(code, code) for code in stock_codes}
    selected_code = st.selectbox(
        '选择个股',
        stock_codes,
        format_func=lambda c: f"{c} {stock_names[c]}",
    )

    if selected_code not in st.session_state.stocks_daily:
        st.error(f"个股 {selected_code} 数据不可用")
        return

    # Load top indicators from backtest results
    if st.session_state.backtest_results is None:
        path = os.path.join(OUTPUT_DIR, 'backtest_results.csv')
        if os.path.exists(path):
            st.session_state.backtest_results = pd.read_csv(path)

    top_n = st.slider('使用排行榜前 N 名指标', 2, 5, 3)

    top_indicators = []
    if st.session_state.backtest_results is not None:
        top = st.session_state.backtest_results.head(top_n)
        for _, row in top.iterrows():
            top_indicators.append(row['指标名'])

    if not top_indicators:
        st.warning("暂无回测结果，请先在标签页2运行回测")
        return

    st.subheader(f"{stock_names[selected_code]} — 当前信号状态")

    # Check daily and weekly
    for freq_label, data_dict in [('日线', st.session_state.stocks_daily),
                                   ('周线', st.session_state.stocks_weekly)]:
        st.markdown(f"**{freq_label}**")
        if selected_code not in data_dict:
            st.caption("无数据")
            continue

        df = data_dict[selected_code].copy()
        df['trade_date'] = pd.to_datetime(df['trade_date'])
        df = df.sort_values('trade_date').reset_index(drop=True)

        results = []
        for ind_name in top_indicators:
            cfg = INDICATOR_REGISTRY.get(ind_name)
            if cfg is None:
                continue

            params = cfg['params'][0]
            try:
                signals = cfg['func'](df, **params)
            except Exception:
                continue

            # Get most recent signals (last 60 trading days)
            recent_cutoff = df['trade_date'].max() - pd.Timedelta(days=120)
            recent_mask = df['trade_date'] >= recent_cutoff
            recent_signals = signals[recent_mask]
            last_signal_date = None
            if recent_signals.any():
                last_signal_date = df['trade_date'][recent_signals].iloc[-1]

            results.append({
                '指标': ind_name,
                '类别': cfg['category'],
                '最近信号日': last_signal_date.date() if last_signal_date is not None else '无',
                '距今(天)': (df['trade_date'].max().date() - last_signal_date.date()).days
                            if last_signal_date is not None else '-',
            })

        if results:
            st.dataframe(pd.DataFrame(results), use_container_width=True, hide_index=True)

    # Overall judgment
    st.divider()
    st.subheader("当前总体判断")

    # Count signals triggered in last 60 days (daily)
    df_daily = st.session_state.stocks_daily[selected_code].copy()
    df_daily['trade_date'] = pd.to_datetime(df_daily['trade_date'])
    df_daily = df_daily.sort_values('trade_date').reset_index(drop=True)
    recent_cutoff = df_daily['trade_date'].max() - pd.Timedelta(days=90)

    triggered = 0
    total_checked = 0
    signal_details = []

    for ind_name in top_indicators:
        cfg = INDICATOR_REGISTRY.get(ind_name)
        if cfg is None:
            continue
        params = cfg['params'][0]
        try:
            signals = cfg['func'](df_daily, **params)
        except Exception:
            continue

        total_checked += 1
        recent = signals & (pd.Series(df_daily['trade_date'].values, index=signals.index) >= recent_cutoff)
        if recent.any():
            triggered += 1
            sig_dates = df_daily['trade_date'][recent.values]
            signal_details.append(f"{ind_name}: {sig_dates.iloc[-1].date()}")

    if total_checked > 0:
        score_pct = triggered / total_checked * 100
        if score_pct >= 60:
            st.success(f"信号共振: {triggered}/{total_checked} 指标已触发 ({score_pct:.0f}%) — 关注买入机会")
        elif score_pct >= 30:
            st.warning(f"部分信号: {triggered}/{total_checked} 指标已触发 ({score_pct:.0f}%) — 继续观察")
        else:
            st.info(f"信号稀少: {triggered}/{total_checked} 指标已触发 ({score_pct:.0f}%) — 等待信号")

        if signal_details:
            with st.expander("信号详情"):
                for d in signal_details:
                    st.write(f"- {d}")
    else:
        st.info("无法判断：所选指标在当前个股上计算失败")
```
